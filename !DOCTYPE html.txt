        function simulateMatch() {
            var player1 = document.getElementById('player1').value;
            var player2 = document.getElementById('player2').value;
            var style1 = document.getElementById('style1').value;
            var style2 = document.getElementById('style2').value;

            var result = "Trận đấu giữa " + player1 + " (Lối đánh: " + style1 + ") và " + player2 + " (Lối đánh: " + style2 + ") đang diễn ra...<br>";

            // Giả sử phân tích tình huống đơn giản
            if (style1 === 'tấn công' && style2 === 'phòng thủ') {
                result += player1 + " có lợi thế tấn công, hãy tiếp tục áp lực để giành điểm!";
            } else if (style1 === 'phòng thủ' && style2 === 'tấn công') {
                result += player2 + " có lợi thế tấn công, hãy cẩn thận và tìm cơ hội phản công!";
            } else {
                result += "Hai người chơi đang có lối đánh cân bằng, hãy tập trung vào chiến thuật để giành điểm!";
            }

            document.getElementById('analysisResult').innerHTML = result;
        }
